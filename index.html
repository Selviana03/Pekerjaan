Abstract Class (Abstraksi) abstract class Pekerja { String nama; Pekerja(this.nama); // Method Abstract: Wajib diimplementasikan oleh subclass void bekerja(); // Method Non-Abstract void identitas() { print("Karyawan: $nama | Pengawas: Serlin Selviana Giay"); } } // 2. Implementasi Class (Inheritance & Overriding) class Programmer extends Pekerja { Programmer(String nama) : super(nama); @override void bekerja() => print("$nama sedang mengembangkan sistem perbankan."); } class Dokter extends Pekerja { Dokter(String nama) : super(nama); @override void bekerja() => print("$nama sedang memberikan layanan kesehatan."); } class Guru extends Pekerja { Guru(String nama) : super(nama); @override void bekerja() => print("$nama sedang memberikan pelatihan teknis."); } // 4. Multiple Interface Simulation (Tablet) class Komputer { void proses() => print("Memproses data..."); } class Kamera { void foto() => print("Mengambil foto..."); } class Telepon { void call() => print("Menghubungi nasabah..."); } class Tablet implements Komputer, Kamera, Telepon { @override void proses() => print("Tablet NIM 251410030: Memproses data."); @override void foto() => print("Tablet NIM 251410030: Foto dokumen."); @override void call() => print("Tablet NIM 251410030: Telepon support."); } // 5. MathUtils (Static Methods) class MathUtils { static int factorial(int n) => (n <= 1) ? 1 : n * factorial(n - 1); static bool isPrima(int n) { if (n < 2) return false; for (int i = 2; i <= sqrt(n); i++) { if (n % i == 0) return false; } return true; } static double hitungRataRata(List angka) { return angka.isEmpty ? 0 : angka.reduce((a, b) => a + b) / angka.length; } } // --- MAIN FUNCTION --- void main() { print("========================================"); print("NAMA : SERLIN SELVIANA GIAY"); print("NIM : 251410030"); print("========================================\n"); // 3. Demonstrasi Polimorfisme List daftar = [ Programmer("Budi"), Dokter("Siti"), Guru("Andi") ]; print("-- Output Polimorfisme --"); for (var p in daftar) { p.identitas(); p.bekerja(); print("--------------------"); } print("\n-- Output MathUtils (Static) --"); print("Faktorial dari 5: ${MathUtils.factorial(5)}"); print("Apakah 23 Bilangan Prima? ${MathUtils.isPrima(23)}"); print("Rata-rata Nilai: ${MathUtils.hitungRataRata([80, 90, 75, 100])}"); }
